// مثال للبحث عن أطول كلمة من حيث عدد الحروف
متغير أطول_كلمة ؛
متغير عدد_الحروف =0؛

متغير الكلمات = مصفوفة(4)؛
الكلمات.تعيين(0,"دولة قطر")؛
الكلمات.تعيين(1,"دولة الامارات العربية المتحدة")؛
الكلمات.تعيين(2,"المملكة العربية السعودية")؛
الكلمات.تعيين(3,"جمهورية مصر العربية")؛

منذ(متغير ك=0؛ك<الكلمات .الطول؛ك++)
{
متغير م=الكلمات.العنصر(ك)؛
	عندما(نص.الطول(م)> عدد_الحروف)
	{
		عدد_الحروف=نص.الطول(م)؛
		أطول_كلمة = الكلمات.العنصر(ك)؛
	}
}

اطبع("أطول كلمة هي: " + أطول_كلمة)؛
اطبع_سطر("عدد الحروف: " +عدد_الحروف)؛
