// مثال لمعرفة عدد الكلمات في جملة
متغير المقولة ="لغتنا تحتاج إلى مبادرات لا إلى محاضرات - صاحب السمو الشيخ محمد بن راشد آل مكتوم"؛
متغير عدد_الكلمات= 0؛
متغير الترتيب =0؛

طالما( الترتيب <نص.الطول(المقولة))
{
	طالما( الترتيب <نص.الطول(المقولة) و نص.الحروف( المقولة,الترتيب ,1) != " " )
	{
		الترتيب++؛
	}
	
	عدد_الكلمات++؛
	طالما( الترتيب <نص.الطول(المقولة) و نص.الحروف( المقولة,الترتيب ,1) == " " )
	{
		الترتيب++؛
	}
}

اطبع("عدد الكلمات: " +عدد_الكلمات )؛
