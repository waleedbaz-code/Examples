// التحقق من الكلمة هل ينطبق عليها السياق المتناظر
// تطابق الكلمة من اليمين إلى اليسار وبالعكس 

مهمة  سياق_متناظر(الكلمة) {
    
  
  متغير ط = نص.الطول(الكلمة)؛ // طول الكلمة
    
	// الدوران على نصف الكلمة
	منذ(متغير ك=0؛ك<ط/2؛ك++)
        // هل يتطابق آخر وأول حرف 
        عندما (نص.الحروف(الكلمة,ك,1) != نص.الحروف(الكلمة,ط-ك-1,1) )
			ارجاع الكلمة + ": سياق غير متناظر"؛

            //في حال التطابق
    ارجاع الكلمة + ": سياق متناظر"؛
}

اطبع_سطر(سياق_متناظر("باب"))؛
اطبع_سطر(سياق_متناظر("فلك"))؛
