	 _.الماركة = ماركة؛
	_.الموديل = موديل؛
	_.كهربائي =هل_كهربائي؛
 	}

	{
		ارجاع  التاريخ().السنة - _.الموديل؛
	}

	{
		عندما(_.كهربائي == متحقق)
			ارجاع "سيارة كهربائية"؛	
		غيرذلك
			ارجاع "سيارة غير كهربائية"؛	
	}
}
